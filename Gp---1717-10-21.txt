DELIMITER //

CREATE PROCEDURE getstudentresults(
    IN p_student_id VARCHAR(20)
)
BEGIN
    SELECT
            Result_Id,
             student_id,
    course_code,
             Semester,
            CA_Marks,
             Final_Marks,
             Final_Grade,
             Result_Status,
             SGPA,
           CGPA
         FROM student_result
         WHERE student_id = p_student_id
        ORDER BY course_code;
    
END //

DELIMITER ;



call getstudentresults('tg/2023/0002');
